#!/bin/bash
sudo yum update -y
sudo yum install git -y
sudo yum install gcc -y

cd /home/ec2-user

sudo curl https://www.sqlite.org/2020/sqlite-autoconf-3320300.tar.gz
sudo tar xzf - sqlite-autoconf-3320300.tar.gz
sudo cd ./sqlite-autoconf-3320300
sudo ./configure
sudo make
sudo make install
cd ..
#sudo rm -rf ./sqlite-autoconf-3320300
#sudo rm sqlite-autoconf-3320300.tar.gz

git config --global user.name "tibbben"
git config --global user.email tibben@ocf.berkeley.edu
git config --global credential.helper store

sudo amazon-linux-extras install docker
sudo usermod -a -G docker ec2-user
sudo service docker start
sudo curl -L https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m) -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

git clone https://github.com/tibbben/pg_tileserv.git
cd pg_tileserv/examples/docker
sudo /usr/local/bin/docker-compose build
sudo /usr/local/bin/docker-compose up

sudo ./load-data.sh

sudo /usr/local/bin/docker-compose down
sudo /usr/local/bin/docker-compose up
